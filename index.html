const CURRENCIES = [

  { code: 'GBP', name: 'Libra', flag: '🇬🇧' },

  { code: 'USD', name: 'Dólar', flag: '🇺🇸' },

  { code: 'CHF', name: 'Franco', flag: '🇨🇭' },

  { code: 'NOK', name: 'C. Noruega', flag: '🇳🇴' },

];


const DEFAULT_RATES = {

  GBP: 0.8520,

  USD: 1.0850,

  CHF: 0.9450,

  NOK: 11.6500,

  SEK: 11.2800,

  DKK: 7.4580,

  PLN: 4.3200,

  MAD: 10.8500

};


const STORAGE_KEY = 'ritcher_ecb_rates';

const BCE_URL = 'https://www.ecb.europa.eu/stats/policy_and_exchange_rates/euro_reference_exchange_rates/html/index.en.html';


function CurrencyConverter() {

  const [rates, setRates] = useState(DEFAULT_RATES);

  const [lastUpdate, setLastUpdate] = useState(null);

  const [selectedCurrency, setSelectedCurrency] = useState('GBP');

  const [foreignAmount, setForeignAmount] = useState(100);

  const [margin, setMargin] = useState(5);

  const [customRate, setCustomRate] = useState(null);

  const [editingRate, setEditingRate] = useState(false);

  const [tempRate, setTempRate] = useState('');

  const [editingRates, setEditingRates] = useState(false);

  const [draftRates, setDraftRates] = useState(DEFAULT_RATES);


  useEffect(function () {

    try {

      const saved = localStorage.getItem(STORAGE_KEY);

      if (saved) {

        const data = JSON.parse(saved);

        setRates(data.rates);

        setLastUpdate(data.lastUpdate);

        setDraftRates(data.rates);

      }

    } catch (err) {}

  }, []);


  useEffect(function () {

    setCustomRate(null);

    setEditingRate(false);

  }, [selectedCurrency, margin]);


  const eurToForeign = rates[selectedCurrency] || 0;

  const officialRateForeignToEur = eurToForeign > 0 ? 1 / eurToForeign : 0;

  const sellingRateForeignToEur = customRate !== null

    ? customRate

    : officialRateForeignToEur * (1 - margin / 100);

  const officialEurAmount = foreignAmount * officialRateForeignToEur;

  const eurToGive = foreignAmount * sellingRateForeignToEur;

  const profitEur = officialEurAmount - eurToGive;

  const effectiveMargin = officialRateForeignToEur > 0

    ? ((1 - sellingRateForeignToEur / officialRateForeignToEur) * 100)

    : 0;


  const currentCurrency = CURRENCIES.find(function (c) { return c.code === selectedCurrency; });


  function formatNumber(num, decimals) {

    const d = decimals == null ? 2 : decimals;

    return num.toLocaleString('es-ES', { minimumFractionDigits: d, maximumFractionDigits: d });

  }


  function formatDate(isoString) {

    if (!isoString) return 'Sin actualizar';

    const d = new Date(isoString);

    return d.toLocaleDateString('es-ES', { day: '2-digit', month: '2-digit', year: 'numeric' });

  }


  function daysSinceUpdate() {

    if (!lastUpdate) return null;

    return Math.floor((Date.now() - new Date(lastUpdate).getTime()) / 86400000);

  }


  function handleEditRate() {

    setTempRate(sellingRateForeignToEur.toFixed(4));

    setEditingRate(true);

  }


  function handleSaveRate() {

    const r = parseFloat(tempRate);

    if (!isNaN(r) && r > 0) setCustomRate(r);

    setEditingRate(false);

  }


  function resetCustomRate() { setCustomRate(null); }


  function startEditingRates() {

    setDraftRates(Object.assign({}, rates));

    setEditingRates(true);

  }


  function cancelEditingRates() {

    setDraftRates(Object.assign({}, rates));

    setEditingRates(false);

  }


  function saveRates() {

    const newLastUpdate = new Date().toISOString();

    try {

      localStorage.setItem(STORAGE_KEY, JSON.stringify({ rates: draftRates, lastUpdate: newLastUpdate }));

      setRates(Object.assign({}, draftRates));

      setLastUpdate(newLastUpdate);

      setEditingRates(false);

    } catch (err) {

      alert('Error al guardar las tasas. Intenta de nuevo.');

    }

  }


  function updateDraftRate(code, value) {

    const next = Object.assign({}, draftRates);

    next[code] = parseFloat(value) || 0;

    setDraftRates(next);

  }


  const days = daysSinceUpdate();

  const stale = days !== null && days >= 2;


  return (

    <div className="min-h-screen p-3 sm:p-4 md:p-6">

      <div className="max-w-3xl mx-auto space-y-3 sm:space-y-4">


        {/* HEADER */}

        <div className="bg-white rounded-2xl shadow-md border-2 border-black p-4 sm:p-6">

          <div className="flex items-center justify-between gap-2">

            <div className="flex items-center gap-2.5 min-w-0">

              <div className="bg-yellow-400 p-2 rounded-xl border-2 border-black shrink-0 text-black text-xl">

                <SvgIcon paths={ICONS.calculator} size={22} />

              </div>

              <div className="min-w-0">

                <h1 className="text-base sm:text-xl md:text-2xl font-black text-black tracking-tight leading-tight">

                  Calculadora de Cambio

                </h1>

                <p className="text-xs sm:text-sm font-bold text-blue-700">Ritcher</p>

              </div>

            </div>

            <button

              onClick={startEditingRates}

              className="flex items-center gap-1.5 px-3 py-2 bg-blue-700 active:bg-blue-800 text-white rounded-lg border-2 border-black font-bold shrink-0 min-h-[44px]"

            >

              <SvgIcon paths={ICONS.pencil} size={16} />

              <span className="text-xs sm:text-sm">Tasas BCE</span>

            </button>

          </div>

          <div className={"flex items-center justify-between mt-3 flex-wrap gap-2 p-2 rounded-lg " + (stale ? "bg-yellow-100 border border-yellow-400" : "")}>

            <p className="text-xs text-slate-700 font-medium leading-tight">

              <span className="block sm:inline">Tasas BCE: <span className="font-black text-black">{formatDate(lastUpdate)}</span></span>

              {stale ? <span className="block sm:inline sm:ml-2 font-bold text-yellow-900">⚠ Recomendable actualizar</span> : null}

            </p>

          </div>

        </div>


        {/* PANEL DE EDICIÓN DE TASAS */}

        {editingRates ? (

          <div className="bg-white rounded-2xl shadow-lg border-2 border-blue-700 p-4 sm:p-6">

            <div className="flex items-center justify-between mb-3 gap-2">

              <h2 className="text-base sm:text-lg font-black text-black uppercase tracking-tight">

                Actualizar tasas BCE

              </h2>

              <a

                href={BCE_URL}

                target="_blank"

                rel="noopener noreferrer"

                className="text-xs font-bold text-blue-700 active:text-blue-900 flex items-center gap-1 px-2 py-1.5 bg-blue-50 rounded border border-blue-200 min-h-[36px]"

              >

                <span>Abrir BCE</span>

                <SvgIcon paths={ICONS.external} size={12} />

              </a>

            </div>

            <p className="text-xs sm:text-sm text-slate-600 mb-3">

              Cuánto equivale a <span className="font-bold">1 €</span> según el BCE.

            </p>

            <div className="space-y-2 mb-4">

              {CURRENCIES.map(function (c) {

                return (

                  <div key={c.code} className="flex items-center gap-2 p-2.5 bg-slate-50 rounded-xl border border-slate-200">

                    <span className="text-xl">{c.flag}</span>

                    <div className="flex-1 min-w-0">

                      <div className="font-black text-sm text-black">{c.code}</div>

                      <div className="text-xs text-slate-500 truncate">{c.name}</div>

                    </div>

                    <div className="flex items-center gap-1 shrink-0">

                      <span className="text-xs text-slate-500 font-bold">1 € =</span>

                      <input

                        type="number"

                        step="0.0001"

                        inputMode="decimal"

                        value={draftRates[c.code] || ''}

                        onChange={function (ev) { updateDraftRate(c.code, ev.target.value); }}

                        onFocus={function (ev) { ev.target.select(); }}

                        className="w-24 px-2 py-2 font-black text-black border-2 border-black rounded focus:border-blue-700 focus:outline-none focus:ring-2 focus:ring-yellow-400"

                      />

                    </div>

                  </div>

                );

              })}

            </div>

            <div className="flex gap-2">

              <button

                onClick={cancelEditingRates}

                className="flex-1 flex items-center justify-center gap-2 px-4 py-3 bg-white active:bg-slate-100 text-black rounded-lg border-2 border-black font-bold min-h-[48px]"

              >

                <SvgIcon paths={ICONS.close} size={16} />

                <span className="text-sm">Cancelar</span>

              </button>

              <button

                onClick={saveRates}

                className="flex-1 flex items-center justify-center gap-2 px-4 py-3 bg-yellow-400 active:bg-yellow-500 text-black rounded-lg border-2 border-black font-bold min-h-[48px]"

              >

                <SvgIcon paths={ICONS.save} size={16} />

                <span className="text-sm">Guardar</span>

              </button>

            </div>

          </div>

        ) : null}


        {/* SELECTOR DE MONEDA */}

        <div className="bg-white rounded-2xl shadow-md border-2 border-black p-4 sm:p-6">

          <label className="text-xs sm:text-sm font-black text-black uppercase tracking-wide mb-3 block">

            Moneda del cliente

          </label>

          <div className="grid grid-cols-4 gap-1.5 sm:gap-2">

            {CURRENCIES.map(function (c) {

              const active = selectedCurrency === c.code;

              const base = "p-2 sm:p-3 rounded-xl border-2 transition text-center min-h-[64px] flex flex-col items-center justify-center ";

              const cls = base + (active ? "border-black bg-yellow-400 shadow-md" : "border-slate-300 bg-white active:border-blue-700");

              return (

                <button key={c.code} onClick={function () { setSelectedCurrency(c.code); }} className={cls}>

                  <span className="text-2xl leading-none">{c.flag}</span>

                  <div className="font-black text-black text-xs sm:text-sm mt-1">{c.code}</div>

                </button>

              );

            })}

          </div>

        </div>


        {/* IMPORTE */}

        <div className="bg-white rounded-2xl shadow-md border-2 border-black p-4 sm:p-6">

          <label className="text-xs sm:text-sm font-black text-black uppercase tracking-wide mb-3 block">

            {selectedCurrency} que entrega el cliente

          </label>

          <div className="relative">

            <span className="absolute left-3 top-1/2 -translate-y-1/2 text-2xl sm:text-3xl">

              {currentCurrency ? currentCurrency.flag : ''}

            </span>

            <input

              type="number"

              value={foreignAmount}

              onChange={function (ev) { setForeignAmount(parseFloat(ev.target.value) || 0); }}

              onFocus={function (ev) { ev.target.select(); }}

              step="0.01"

              min="0"

              inputMode="decimal"

              className="big-input w-full pl-14 pr-3 py-4 font-black text-black border-2 border-black rounded-xl focus:border-blue-700 focus:outline-none focus:ring-2 focus:ring-yellow-400"

            />

          </div>

          <div className="grid grid-cols-6 gap-1.5 mt-3">

            {[10, 25, 50, 100, 250, 500].map(function (v) {

              return (

                <button

                  key={v}

                  onClick={function () { setForeignAmount(v); }}

                  className="px-1 py-2.5 text-xs font-bold bg-black active:bg-blue-700 rounded-lg text-white min-h-[40px]"

                >

                  {v}

                </button>

              );

            })}

          </div>

        </div>


        {/* MARGEN */}

        <div className="bg-white rounded-2xl shadow-md border-2 border-black p-4 sm:p-6">

          <div className="flex items-center justify-between mb-3">

            <label className="text-xs sm:text-sm font-black text-black uppercase tracking-wide">

              Tu margen

            </label>

            <span className="text-2xl sm:text-3xl font-black text-blue-700">{margin}%</span>

          </div>

          <input

            type="range"

            min="0"

            max="20"

            step="0.5"

            value={margin}

            onChange={function (ev) { setMargin(parseFloat(ev.target.value)); }}

            className="w-full accent-blue-700"

          />

          <div className="flex justify-between text-xs text-slate-500 font-bold mt-1 mb-3">

            <span>0%</span>

            <span>10%</span>

            <span>20%</span>

          </div>

          <div className="grid grid-cols-6 gap-1.5">

            {[0, 3, 5, 8, 10, 15].map(function (v) {

              const active = margin === v && customRate === null;

              const base = "px-1 py-2.5 text-xs font-bold rounded-lg border-2 min-h-[40px] ";

              const cls = base + (active ? "bg-yellow-400 text-black border-black" : "bg-black text-white border-black active:bg-blue-700");

              return (

                <button key={v} onClick={function () { setMargin(v); }} className={cls}>

                  {v}%

                </button>

              );

            })}

          </div>

        </div>


        {/* RESULTADO HERO */}

        <div className="bg-black rounded-2xl p-5 sm:p-6 text-white border-2 border-yellow-400 relative overflow-hidden shadow-xl">

          <div className="absolute top-0 right-0 w-40 h-40 bg-yellow-400 opacity-10 rounded-full -mr-20 -mt-20"></div>

          <div className="text-[10px] sm:text-xs font-black text-yellow-400 uppercase tracking-widest mb-2">

            Le das al cliente

          </div>

          <div className="flex items-baseline gap-2 flex-wrap">

            <span className="text-5xl sm:text-6xl md:text-7xl font-black text-yellow-400 leading-none">

              {formatNumber(eurToGive)}

            </span>

            <span className="text-2xl sm:text-3xl text-white font-black">€</span>

          </div>

          <div className="text-xs sm:text-sm text-slate-300 mt-3 font-medium">

            Por <span className="font-black text-white">{formatNumber(foreignAmount)} {selectedCurrency} {currentCurrency ? currentCurrency.flag : ''}</span>

                    step="0.0001"

                    autoFocus

              <div className="text-sm sm:text-base font-black text-blue-900 leading-tight">{formatNumber(effectiveMargin, 1)}%</div>
